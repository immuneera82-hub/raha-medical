
# أوامر تحديث السيرفر (Final Fix)
# يرجى تنفيذ هذه الأوامر بدقة:

cd raha-medical

# 1. سحب التعديلات (يجب أن تري ملفات يتم تحديثها)
git pull origin main

# 2. حذف الحاويات القديمة تماماً
docker-compose down

# 3. بناء الحاوية من الصفر (مهم جداً لتطبيق هيكلية المجلدات الجديدة)
docker-compose build --no-cache backend

# 4. التشغيل
docker-compose up -d

# 5. مراقبة السجلات للتأكد من عمل السيرفر
docker-compose logs --tail=50 -f backend
